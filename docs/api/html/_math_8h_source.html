302inline bool ctIsPrime(const uint32_t x) {
303 if (x < 2) { return false; /*actually undefined*/ }
304 if (x < 4) { return true; }
305 if ((x % 2) == 0) { return false; }
306 for (uint32_t i = 3; i <= (uint32_t)ctFloorf(ctSqrtf((float)x)); i += 2) {
307 if ((x % i) == 0) { return false; }
308 }
309 return true;
310}
